1function [pks,locs] = findpeaks_fast(y)
 2%custom findpeaks function much faster than stock
 3%pks = y values, locs = x values (indices)
 4    coder.inline('always') %force inline
 5    pks = zeros(1,length(y)); %allocate (faster than growing)
 6    locs = pks;
 7    if(length(y) < 3) %require at least 3 points
 8        return;
 9    end

11    %I couldn't find a way to make this faster
12    %matches stock output well, may have extra peaks at mesas/end
13    for i = 2:length(y)-1 %exclude bounds like stock
14        if((y(i) - y(i-1) > 0) && (y(i+1) - y(i) <= 0))

16            pks(num) = y(i);
17            locs(num) = i;
18        end
19    end
20    pks = pks(1:num); %output right sized array

22end
