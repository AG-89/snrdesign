1function [Eout, Hout] = EH_iterate(E,H,buffer,maxlag,newrange_size)
 2%perform iterative calculation for E & H
 3    buffer_len = length(buffer);
 4    for sN = 0:newrange_size-1 %sample number in range
 5        for L = 1:maxlag %lag
 6            EHC_newterm1 = buffer(buffer_len-sN); %.* ones(1,maxlag);
 7            EHC_newterm2 = buffer(buffer_len-L-sN-0);
 8            EHC_newterm3 = buffer(buffer_len-2*L-sN);
 9            %EHC_newterm4 = EHC_newterm2;
10            E(L) = E(L) + EHC_newterm1.^2 - EHC_newterm3.^2;
11            H(L) = H(L) + EHC_newterm1.*EHC_newterm2 - EHC_newterm2.*EHC_newterm3;
12        end
13    end
14    Eout = E;
15    Hout = H;
16end
